\text{exp amortised reward } R(t) = \langle \sum_{\tau = 0}^\infty \alpha^\tau r(t+\tau) \rangle \\
r(t)+\alpha r(t+1)+\cdot\cdot\cdot  = r(t) + \alpha (r(t+1) + \alpha r(t+2) + \cdot\cdot\cdot ) = r(t) + \alpha R(t+1)
